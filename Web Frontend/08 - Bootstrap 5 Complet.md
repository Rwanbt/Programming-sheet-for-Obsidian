        }, 3000);
      }, 1500);
    }

    form.classList.add('was-validated');
  });
})();

### Checklist de bonne pratique Bootstrap 5

| Point | À vérifier |
|---|---|
| Structure grid | `container > row > col` — jamais sauter un niveau |
| Mobile first | Partir des classes sans suffixe, enrichir avec `-md-`, `-lg-` |
| CDN JS | Utiliser `bootstrap.bundle.min.js` (inclut Popper) |
| Accessibilité | `aria-label` sur les boutons icônes, `aria-hidden` sur les décorations |
| Formulaires | `for` sur `<label>` correspond à `id` sur `<input>` |
| Tooltips/Popovers | Initialiser manuellement via JS (`new bootstrap.Tooltip(el)`) |
| Images responsives | Ajouter `img-fluid` sur toutes les images dans une grille |
| Navbar | `navbar-expand-*` détermine le breakpoint de collapse |
